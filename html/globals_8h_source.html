158void setUpInterrupt(){
159 // counter A has interrupts enabled
160 TIMSK3 |= (1 << OCIE3A);
161
162 // interrupts enbled
163 SREG |= (1<<7);
164
165 TCNT3H = 0;
166

170
171 OCR3A = 32000;
172
173 // make timer work in CTC mode
174 TCCR3A |= (0 << WGM31) | (0 << WGM30);
175 TCCR3A |= (0 << COM3A1) | (1 << COM3A0);
176 TCCR3B |= (0 << WGM33) | (1 << WGM32);
177
178 // set frequency of timer
179 TCCR3B |= (0 << CS32) | (0 << CS31) | (1 << CS30);
180}

183void schedule_sync() {
184 // wait until interrupt
185 while(sFlag == PENDING){
186
187 }
188
189 // update task run time and if task is sleeping
190 // decrease sleeping time, check if task should
191 // be running, and set task to READY if appropriate
192 if(CURRENT_SCHEDULER == SRRI) {
193 for(int i = 0; i < TASK_AMOUNT; i++){
194 //update time the task has been running for
195 timeTask[i] += 2;
196 if(stateTask[i] == SLEEPING){
197 //update time task has been sleeping for
198 sleepTask[i] -= 2;
199 if(sleepTask[i] <= 1){
200 //task is ready to run once more
201 stateTask[i] = READY;
202 }
203 }
204 }
205 } else if(CURRENT_SCHEDULER == DDS) {
206 for(int i = 0; i < TASK_AMOUNT; i++){
207 TaskList[i].time_ran +=2;
208 if(TaskList[i].state == SLEEPING){
209 TaskList[i].time_slept -=2;
210 if(TaskList[i].time_slept <= 1){
211 TaskList[i].state = READY;
212 }
213 }
214 }
215 }
216
217
218
219 sFlag = PENDING;
220 return;
221}
